49  private void Start()
50  {
51  // Set up the limits with absolute values from each system.
52  slTemperatureRange.SetLimits(WorldSystem.AbsoluteTempRange.x, WorldSystem.AbsoluteTempRange.y);
53  slTimeBeforeFluctuation.SetLimits(WorldSystem.AbsoluteFluctuationWaitRange.x, WorldSystem.AbsoluteFluctuationWaitRange.y);
54  slFluctuationDuration.SetLimits(WorldSystem.AbsoluteFluctuationLengthRange.x, WorldSystem.AbsoluteFluctuationLengthRange.y);
55  slFluctuationRange.SetLimits(WorldSystem.AbsoluteFluctuationTemperatureRange.x, WorldSystem.AbsoluteFluctuationTemperatureRange.y);
56  slTemperatureMargin.SetLimits(WorldSystem.AbsoluteAcceptanceRange.x, WorldSystem.AbsoluteAcceptanceRange.y);
57  slACCheckTime.minValue = ACSystem.AbsoluteTemperatureCheckRange.x;
58  slACCheckTime.maxValue = ACSystem.AbsoluteTemperatureCheckRange.y;
59  }
60 
61  public override void InitializeScreen()
62  {
63  // Update the values to the current world.
64  slTemperatureRange.SetValues(WorldSystem.WorldTemperatureRange.x, WorldSystem.WorldTemperatureRange.y);
65  slWorldTemperature.value = WorldSystem.CurrentWorldTemperature;
66 
67  toAllowFluctuation.isOn = WorldSystem.AllowTemperatureFluctuation;
68  lastFluctuationToggle = WorldSystem.AllowTemperatureFluctuation;
69  WorldSystem.ToggleTemperatureFluctuation(false);
70 
71  slTimeBeforeFluctuation.SetValues(WorldSystem.FluctuationWaitRange.x, WorldSystem.FluctuationWaitRange.y);
72  slFluctuationDuration.SetValues(WorldSystem.FluctuationLengthRange.x, WorldSystem.FluctuationLengthRange.y);
73  slFluctuationRange.SetValues(WorldSystem.FluctuationTemperatureRange.x, WorldSystem.FluctuationTemperatureRange.y);
74 
75  slTemperatureMargin.SetValues(WorldSystem.CurrentAcceptanceRange.x, WorldSystem.CurrentAcceptanceRange.y);
76 
77  slACCheckTime.value = ACSystem.TemperatureCheckTimer;
78  }
79 
80  public override IEnumerator HideScreen()
81  {
82  WorldSystem.ToggleTemperatureFluctuation(lastFluctuationToggle);
83  yield return base.HideScreen();
84  }

86  /// <summary>
87  /// A callback function when updating the <see cref="slTemperatureRange"/>. This affects what
88  /// the value of the <see cref="WorldSystem.CurrentWorldTemperature"/> can be.
89  /// </summary>
90  public void UpdateWorldTemperatureSliderBounds()
91  {
92  slWorldTemperature.minValue = slTemperatureRange.Values.minValue;
93  slWorldTemperature.maxValue = slTemperatureRange.Values.maxValue;
94  }
95 
96  /// <summary>
97  /// A function called to submit all changes to the systems again, once the user is done.
98  /// </summary>
99  public void SubmitChanges()
100  {
101  WorldSystem.WorldTemperatureRange = new Vector2(slTemperatureRange.Values.minValue, slTemperatureRange.Values.maxValue);
102  WorldSystem.TargetWorldTemperature = WorldSystem.TargetWorldTemperature;
103  WorldSystem.SetWorldTemperature(slWorldTemperature.value);
104  WorldSystem.ToggleTemperatureFluctuation(toAllowFluctuation.isOn);
105  lastFluctuationToggle = toAllowFluctuation.isOn;
106 
107  WorldSystem.FluctuationWaitRange = new Vector2(slTimeBeforeFluctuation.Values.minValue, slTimeBeforeFluctuation.Values.maxValue);
108  WorldSystem.FluctuationLengthRange = new Vector2(slFluctuationDuration.Values.minValue, slFluctuationDuration.Values.maxValue);
109  WorldSystem.FluctuationTemperatureRange = new Vector2(slFluctuationRange.Values.minValue, slFluctuationRange.Values.maxValue);
110 
111  WorldSystem.CurrentAcceptanceRange = new Vector2(slTemperatureMargin.Values.minValue, slTemperatureMargin.Values.maxValue);
112  ACSystem.TemperatureCheckTimer = slACCheckTime.value;
113 
114  ACSystem.CheckTemperatures();
115  }
116  }
