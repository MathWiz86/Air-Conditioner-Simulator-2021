46  {
47  int index = 0; // A current index for the array.
48 
49  // Constantly update the image.
50  while (true)
51  {
52  // Get the second material (as set on the tv model) and set the texture.
53  Material[] mats = mrenderer.materials;
54  mats[1].mainTexture = tvFrames[index];
55  mrenderer.materials = mats;
56 
57  // Increment the index. If the index is more than the number of frames, wrap to 0.
58  index++;
59  if (index >= tvFrames.Length)
60  index = 0;
61 
62  // Wait a random amount of time before the next swap.
63  yield return new WaitForSeconds(Random.Range(swapTimeRange.x, swapTimeRange.y));
