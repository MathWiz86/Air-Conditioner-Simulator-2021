24  /// <summary>
25  /// A custom attribute which allows for creating a range between two numbers via referencing
26  /// other variables on an object.
27  /// </summary>
28  public class RangeRefAttribute : PropertyAttribute
29  {
30  /// <summary>The path to the variable with the minimum value.</summary>
31  public string min;
32  /// <summary>The path to the variable with the maximum. value.</summary>
33  public string max;
34  /// <summary>A bool determining if the passed-in paths are valid.</summary>
35  public bool isValid { get; private set; }
36 
37  /// <summary>
38  /// The constructor for the <see cref="RangeRefAttribute"/>.
39  /// </summary>
40  /// <param name="min">The path to the variable with the minimum value.</param>
41  /// <param name="max">The path to the variable with the maximum. value.</param>
42  public RangeRefAttribute(string min, string max)
43  {
44  this.min = string.IsNullOrWhiteSpace(min) ? string.Empty : min; // Set the minimum value.
45  this.max = string.IsNullOrWhiteSpace(max) ? string.Empty : max; // Set the maximum value.
46 
47  // The attribute is considered valid if both the min and max start with the variable call character.
48  isValid = min.StartsWith(Kits.Reflection.VariableCall) && max.StartsWith(Kits.Reflection.VariableCall);
49  }
50  }

130 
131  // Finish clamping values.

135  value.y = Mathf.Clamp(value.y, minValue, maxValue);
136  value.x = Mathf.Clamp(value.x, minValue, value.y);

138  }

200  public override void OnGUI(Rect position, SerializedProperty property, GUIContent label)
201  {
202  RangeRefAttribute attribute = this.attribute as RangeRefAttribute;
203  valuesFound = attribute.isValid;
204 
205  // If this attribute is valid, meaning it targets variables, check their dynamic field values.
206  if (attribute.isValid)
207  {
208  valuesFound = valuesFound & Reflection.CheckDynamicField(property.serializedObject.targetObject, attribute.min, minValue, out minValue);
209  valuesFound = valuesFound & Reflection.CheckDynamicField(property.serializedObject.targetObject, attribute.max, maxValue, out maxValue);
210  }
211 
212  // Perform a different draw function depending on the property type.
213  switch (property.propertyType)
214  {
215  case SerializedPropertyType.Integer:
216  OnPropertyInt(position, property, label);
217  break;
218  case SerializedPropertyType.Float:
219  OnPropertyFloat(position, property, label);
220  break;
221  case SerializedPropertyType.Vector2:
222  OnPropertyVector2(position, property, label);
223  break;
224  case SerializedPropertyType.Vector2Int:
225  OnPropertyVector2Int(position, property, label);
226  break;
227  default:
228  EditorGUI.PropertyField(position, property, label);
229  break;
230  }
231  }
