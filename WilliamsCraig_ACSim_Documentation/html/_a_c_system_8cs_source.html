26 /// <summary>
27 /// The singleton system for the A/C unit. This primarily is for handling changing the temperature
28 /// based on fuzzy logic.
29 /// </summary>
30 public class ACSystem : MonoBehaviour
31 {
32  /// <summary>The singleton signature for this system.</summary>
33  private static ACSystem _singleton = null;
34  /// <summary>The absolute range of time allowed between each temperature check.</summary>
35  public static readonly Vector2 AbsoluteTemperatureCheckRange = new Vector2(0.0f, 15.0f);
36 
37  /// <summary>The amount of time waited between every check of the world temperature vs the target temperature.</summary>
38  public static float TemperatureCheckTimer { get { return _singleton._TemperatureCheckTimer; } set { _singleton._TemperatureCheckTimer = Mathf.Clamp(value, AbsoluteTemperatureCheckRange.x, AbsoluteTemperatureCheckRange.y); } }
39  /// <summary>The internal value for <see cref="TemperatureCheckTimer"/>.</summary>
40  [SerializeField] [RangeRef("!AbsoluteTemperatureCheckRange.x", "!AbsoluteTemperatureCheckRange.y")]
41  private float _TemperatureCheckTimer = 10.0f;
42 
43  /// <summary>The antecedent for 'Much Lower'.</summary>
44  private readonly TSTriangleMembershipFunction antecedent1 = new TSTriangleMembershipFunction((f) => -6.0f * f + 1);
45  /// <summary>The antecedent for 'Lower'.</summary>
46  private readonly TSTriangleMembershipFunction antecedent2 = new TSTriangleMembershipFunction((f) => -2.5f * f);
47  /// <summary>The antecedent for 'Acceptable'.</summary>
48  private readonly TSTriangleMembershipFunction antecedent3 = new TSTriangleMembershipFunction((f) => 0.0f);
49  /// <summary>The antecedent for 'Higher'.</summary>
50  private readonly TSTriangleMembershipFunction antecedent4 = new TSTriangleMembershipFunction((f) => 2.5f * f);
51  /// <summary>The antecedent for 'Much Higher.</summary>
52  private readonly TSTriangleMembershipFunction antecedent5 = new TSTriangleMembershipFunction((f) => 6.0f * f + 1);
53 
54  /// <summary>The fan object that will spin in tune with the temperature speed.</summary>
55  [SerializeField] private ACFan acFan = null;
56  /// <summary>The change in temperature the world will have per second while the fan is running.</summary>
57  private float temperatureChangeSpeed = 0.0f;
58  /// <summary>The <see cref="Coroutine"/> for awaiting the temperature check.</summary>
59  private Coroutine coAwaitTemperatureCheck = null;
60  /// <summary>The <see cref="Coroutine"/> for changing the temperature.</summary>
61  private Coroutine coChangeTemperature = null;

81  /// <summary>
82  /// A function to force the temperatures to be rechecked. Typically, this is called when
83  /// world values are updated.
84  /// </summary>
85  public static void CheckTemperatures()
86  {

88  _singleton.CheckTemperaturesInternal();

91  /// <summary>
92  /// The function that handles checking if the A/C unit needs to be turned on and change
93  /// the world temperature.
94  /// </summary>
95  private void CheckTemperaturesInternal()
96  {
97  UpdateAntecedents(); // Update the antecedent values.
98 
99  // Get the difference in temperature between the target and current.
100  float tempDifference = WorldSystem.TargetWorldTemperature - WorldSystem.CurrentWorldTemperature;
101  // Get the amount of change in the speed.
102  float tempChange = FuzzyKit.CalculateTakagiSugenoSingle(tempDifference, antecedent1, antecedent2, antecedent3, antecedent4, antecedent5);
103  InformationScreen.UpdateTSOutputValueDisplay(tempChange);
104 
105  // If the change is not 0, we need to turn on the fan.
106  if (tempChange != 0.0f)
107  {
108  StopCoroutine(coAwaitTemperatureCheck); // Stop the check timer.
109  temperatureChangeSpeed = tempChange; // Set the change speed.
110 
111  if (coChangeTemperature != null)
112  StopCoroutine(coChangeTemperature);
113 
114  coChangeTemperature = StartCoroutine(HandleACTemperatureChange()); // Start changing the temperature.
115  }
116  }
117 
118  /// <summary>
119  /// A function which updates the antecedent values depending on the current world temperature settings.
120  /// </summary>
121  private void UpdateAntecedents()
122  {
123  // Get the max range between the world temperature values.
124  float maxDifference = WorldSystem.WorldTemperatureRange.y - WorldSystem.WorldTemperatureRange.x;
125 
126  // Update the values based on the current difference and current acceptance range.
127  antecedent1.ResetValues(-maxDifference, -maxDifference, -maxDifference * 0.3f);
128  antecedent2.ResetValues(-maxDifference * 0.5f, -maxDifference * 0.3f, WorldSystem.CurrentAcceptanceRange.x);
129  antecedent3.ResetValues(WorldSystem.CurrentAcceptanceRange.x, 0.0f, WorldSystem.CurrentAcceptanceRange.y);
130  antecedent4.ResetValues(WorldSystem.CurrentAcceptanceRange.y, maxDifference * 0.3f, maxDifference * 0.5f);
131  antecedent5.ResetValues(maxDifference * 0.3f, maxDifference, maxDifference);
132 
133  InformationScreen.UpdateMembershipFunctionDisplays(antecedent1, antecedent2, antecedent3, antecedent4, antecedent5);
134  }
135 
136  /// <summary>
137  /// A simple timer function to wait before checking the temperature.
138  /// </summary>
139  /// <returns>Returns an IEnumerator to use in <see cref="Coroutine"/>s.</returns>
140  private IEnumerator AwaitTemperatureCheck()
141  {
142  while (true)
143  {
144  // Until we can change temperature, wait the alotted time before checking the temperatures again.
145  yield return new WaitForSeconds(_TemperatureCheckTimer);
146  CheckTemperaturesInternal();
147  }
148  }
149 
150  /// <summary>
151  /// An IEnumerator for handling the A/C, and changing the world temperature based on the fuzzy system results.
152  /// </summary>
153  /// <returns>Returns an IEnumerator to use in <see cref="Coroutine"/>s.</returns>
154  private IEnumerator HandleACTemperatureChange()
155  {
156  WorldSystem.ToggleActiveTemperatureFluctuation(false); // The world should not fluctuate during this portion.
157  yield return acFan.StartSpin(temperatureChangeSpeed); // Spin up the fan.
158 
159  // Get the sign of what direction we are going: Heating or Cooling.
160  int sign = System.Math.Sign(WorldSystem.CurrentWorldTemperature - WorldSystem.TargetWorldTemperature);
161  InformationScreen.UpdateAirConditionerStateDisplay(sign < 0 ? InformationScreen.ACState.Heating : InformationScreen.ACState.Cooling);
162  // While we are still changing, update the temperature every second.
163  while ((sign * WorldSystem.CurrentWorldTemperature) > (sign * WorldSystem.TargetWorldTemperature))
164  {
165  WorldSystem.ChangeWorldTemperature(temperatureChangeSpeed * Time.deltaTime);
166  yield return null;
167  }
168 
169  yield return acFan.EndSpin(); // Stop the fan.
170  InformationScreen.UpdateAirConditionerStateDisplay(InformationScreen.ACState.Off);
171  WorldSystem.ToggleActiveTemperatureFluctuation(true); // The world can fluctuate again.
172  coAwaitTemperatureCheck = StartCoroutine(AwaitTemperatureCheck()); // Start waiting for a change again.
173 
174  yield return null;
175  }
176 }
