35  /// <summary>

38  /// <returns>Returns an <see cref="IEnumerator"/> to use in <see cref="Coroutine"/>s.</returns>
39  private IEnumerator HandleTitleScreen()
40  {
41  yield return AwaitAnimation("DISPLAY"); // Display the title screen.
42  yield return new WaitUntil(() => Input.GetKeyDown(KeyCode.Mouse0)); // Wait for a mouse click.
43  yield return AwaitAnimation("HIDE"); // Hide the title screen.
44 
45  WorldSystem.BeginSimulation(); // Begin the simulation.
46  Destroy(this.gameObject); // The title screen is no longer needed.
47  }

49  /// <summary>

51  /// </summary>
52  /// <param name="animName">The name of the animation state.</param>
53  /// <param name="layer">The layer of the animation state.</param>
54  /// <returns>Returns an <see cref="IEnumerator"/> to use in <see cref="Coroutine"/>s.</returns>
55  protected IEnumerator AwaitAnimation(string animName, int layer = 0)
56  {
57  titleAnimator.Play(animName, layer); // Play the animation.
58 
59  // Wait a bit, to make sure the state updates. Then get the time remaining.
60  yield return new WaitForSeconds(0.01f);
61  float time = titleAnimator.GetCurrentAnimatorStateInfo(0).length - 0.01f;
62  // Wait out the remaining time.
63  yield return new WaitForSeconds(time);
