76  /// <summary>
77  /// A function to update current target value.
78  /// </summary>
79  /// <param name="change">The amount to change the temperature by.</param>
80  private void UpdateTargetWorldTemperature(float change)
81  {
82  // Update the current target, while keeping on the world range.
83  Vector2 range = WorldSystem.WorldTemperatureRange;
84  currentTarget = Mathf.Clamp(currentTarget + change, range.x,range.y);
85 
86  // Update the text and fill amount.
87  tmpTargetTemperature.text = currentTarget.ToString();
88  imgTemperatureGauge.fillAmount = (currentTarget - range.x) / (range.y - range.x);
89 
90  // Update the buttons to only be enabled or disabled when not at the extremities of the range.
91  buDecrement.IsEnabled = currentTarget > range.x;
92  buIncrement.IsEnabled = currentTarget < range.y;
93  }

95  /// <summary>
96  /// A routine for changing the temperature and holding the button.
97  /// </summary>
98  /// <param name="change">The amount to change the temperature by.</param>
99  /// <param name="button">The button that caused the change.</param>
100  /// <returns>Returns an <see cref="IEnumerator"/> to use in <see cref="Coroutine"/>s.</returns>
101  private IEnumerator HoldChangeButton(float change, MenuButton button)
102  {
103  StartCoroutine(ChangeButtonScale(button)); // Change the button scale once.
104  UpdateTargetWorldTemperature(change); // Update the temperature once.
105 
106  // Wait a bit of time. If this function is still active by that point, we are holding the button.
107  yield return new WaitForSeconds(timerHoldWait);
108  isHoldingButton = true;
109 
110  // Start changing the button scale again.
111  StartCoroutine(ChangeButtonScale(button));
112 
113  // Until the button is no longer held, update the target temperature every increment.
114  while (true)
115  {
116  UpdateTargetWorldTemperature(change);
117  yield return new WaitForSeconds(timerHoldChange);
118  }
119  }

121  /// <summary>
122  /// A function to animate the button growing and shrinking each click.
123  /// </summary>
124  /// <param name="button">The button that caused the change.</param>
125  /// <returns>Returns an <see cref="IEnumerator"/> to use in <see cref="Coroutine"/>s.</returns>
126  private IEnumerator ChangeButtonScale(MenuButton button)
127  {
128  // Get information on the rect transform and the start and target scales.
129  RectTransform buRTR = button.GetComponent<RectTransform>();
130  Vector3 start = buRTR.localScale;
131  Vector3 target = new Vector3(start.x * maxButtonScale.x, start.y * maxButtonScale.y, start.z * maxButtonScale.z);
132 
133  do
134  {
135  float lerpTimer = 0.0f;
136  // While lerping, update the scale to grow up.
137  while (lerpTimer < timerButtonGrowth)
138  {
139  lerpTimer += Time.deltaTime;
140  buRTR.localScale = Vector3.Lerp(start, target, lerpTimer / timerButtonGrowth);
141  yield return null;
142  }
143 
144  lerpTimer = 0.0f;
145  // While lerping, update the scale to grow down.
146  while (lerpTimer < timerButtonGrowth)
147  {
148  lerpTimer += Time.deltaTime;
149  buRTR.localScale = Vector3.Lerp(target, start, lerpTimer / timerButtonGrowth);
150  yield return null;
151  }
152  }
153  while (isHoldingButton); // Do this while holding the button.
154  }
