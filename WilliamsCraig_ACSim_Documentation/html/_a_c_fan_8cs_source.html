21  /// <summary>
22  /// An air conditioner object. This handles the speed of the fan based on how fast the
23  /// world temperature is changing.
24  /// </summary>
25  public class ACFan : MonoBehaviour
26  {
27  /// <summary>The <see cref="Transform"/> of the fan. This will be spun.</summary>
28  [SerializeField] private Transform fanTR = null;
29 
30  /// <summary>The current speed the fan is spinning.</summary>
31  private float currentSpeed = 0.0f;
32  /// <summary>The target speed the fan will reach.</summary>
33  private float targetSpeed = 0.0f;
34  /// <summary>The time it takes for the fan to spin up to max speed.</summary>
35  private float spinUpTime = 4.1f;
36  /// <summary>The time it takes for the fan to spin down to a standstill.</summary>
37  private float spinDownTime = 4.2f;
38 
39  /// <summary>The range of speed that the fan can spin.</summary>
40  private Vector2 spinSpeedRange = new Vector2(2.0f, 3.5f);
41 
42  /// <summary>The coroutine for spinning up to the <see cref="targetSpeed"/>.</summary>
43  private Coroutine coSpinToTarget;
44  /// <summary>The coroutine for spinning the fan constantly.</summary>
45  private Coroutine coSpinConstantly;
46 
47  /// <summary>
48  /// A function to update the current rotation of the fan, based on the speed.
49  /// </summary>
50  private void UpdateRotation()
51  {
52  Vector3 angles = fanTR.localEulerAngles; // Get the current rotation.
53  angles.z += (currentSpeed * 400 * Time.deltaTime); // Add the next rotation.
54  fanTR.localEulerAngles = angles; // Set the new rotation.
55  }
56 
57  /// <summary>
58  /// A function called to start spinning the fan.
59  /// </summary>
60  /// <param name="target">The target speed to spin up to.</param>
61  /// <returns>Returns a Coroutine that matches <see cref="coSpinToTarget"/>.</returns>
62  public Coroutine StartSpin(float target)
63  {
64  // Stop the coroutines that are currently happening.
65  if (coSpinToTarget != null)
66  StopCoroutine(coSpinToTarget);
67  if (coSpinConstantly != null)
68  StopCoroutine(coSpinConstantly);
69 
70  // Calculate the target speed. It should be within range, and in the direction of heating or cooling.
71  targetSpeed = Mathf.Clamp(Mathf.Abs(target), spinSpeedRange.x, spinSpeedRange.y);
72  targetSpeed *= Mathf.Sign(target);
73 
74  // Start spinning to the target speed. Then, spin constantly.
75  coSpinToTarget = StartCoroutine(SpinToTargetSpeed(targetSpeed, spinUpTime, () => coSpinConstantly = StartCoroutine(SpinFan())));
76  return coSpinToTarget;
77  }
78 
79  /// <summary>
80  /// A function to end the current fan spin.
81  /// </summary>
82  /// <returns>Returns a Coroutine that matches <see cref="coSpinToTarget"/>.</returns>
83  public Coroutine EndSpin()
84  {
85  // Stop the coroutines that are currently happening.
86  if (coSpinToTarget != null)
87  StopCoroutine(coSpinToTarget);
88 
89  if (coSpinConstantly != null)
90  StopCoroutine(coSpinConstantly);
91 
92  // Start spinning down to 0 speed.
93  coSpinToTarget = StartCoroutine(SpinToTargetSpeed(0.0f, spinDownTime));
94  return coSpinToTarget;
95  }
96 
97  /// <summary>
98  /// A routine for spinning the fan up to a target speed.
99  /// </summary>
100  /// <param name="target">The target speed to reach.</param>
101  /// <param name="time">The amount of time to take to reach the <paramref name="target"/>.</param>
102  /// <param name="endTask">An ending task to call once the fan is up to speed.</param>
103  /// <returns>Returns a <see cref="IEnumerator"/>, which can be used in a <see cref="Coroutine"/>.</returns>
104  public IEnumerator SpinToTargetSpeed(float target, float time, System.Action endTask = null)
105  {
106  float lerpTimer = 0.0f;
107  float startSpeed = currentSpeed;
108 
109  // While lerping, update the current speed and rotate the fan.
110  while (lerpTimer < time)
111  {
112  lerpTimer += Time.deltaTime;
113 
114  currentSpeed = Mathf.Lerp(startSpeed, target, lerpTimer / time);
115  UpdateRotation();
116  yield return null;
117  }
118 
119  endTask?.Invoke(); // Invoke the final task.
120  }
121 
122  /// <summary>
123  /// A function which constantly spins the fan at the current speed.
124  /// </summary>
125  /// <returns>Returns a <see cref="IEnumerator"/>, which can be used in a <see cref="Coroutine"/>.</returns>
126  private IEnumerator SpinFan()
127  {
128  // Until manually stopped, update the rotation.
129  while (true)
130  {
131  UpdateRotation();
132  yield return null;
133  }
134  }
135  }
136 }
