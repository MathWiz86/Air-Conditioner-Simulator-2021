106  {
107  raycastBlocker.raycastTarget = true; // Block any mouse clicks.
108 
109  // Play the hide animation for the previous screen.
110  if (hideScreen != null)
111  {
112  yield return StartCoroutine(hideScreen.HideScreen());
113  hideScreen.screenCanvas.enabled = false;
114  }
115 
116  // Play the display animation for the current screen.
117  if (displayScreen != null)
118  {
119  displayScreen.InitializeScreen();
120  displayScreen.screenCanvas.enabled = true;
121  yield return StartCoroutine(displayScreen.DisplayScreen());
122  }
123 
124  raycastBlocker.raycastTarget = false; // Allow mouse clicks.
125  }
126  }
