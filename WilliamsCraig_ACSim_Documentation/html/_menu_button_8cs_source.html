140  /// <summary>
141  /// An effect for moving the button to a specific position.
142  /// </summary>
143  /// <param name="position">The distance to move the button.</param>
144  /// <param name="time">The amount of time to take to move the button.</param>
145  public void MoveButtonPositionTo(Vector2 position, float time = 0.4f)
146  {
147  // If we can move, reset the coroutine and move again.
148  if (moveRTR)
149  {
150  if (coMoveRTR != null)
151  StopCoroutine(coMoveRTR);
152 
153  coMoveRTR = StartCoroutine(HandlePositionChange(position, time));
154  }
155  }

192  /// <summary>
193  /// A routine for moving the button to a new position.
194  /// </summary>
195  /// <param name="position">The distance to move the button.</param>
196  /// <param name="time">The amount of time to take to move the button.</param>
197  /// <returns>Returns an <see cref="IEnumerator"/> to use in <see cref="Coroutine"/>s.</returns>
198  private IEnumerator HandlePositionChange(Vector2 position, float time)
199  {
200  Vector2 start = moveRTR.anchoredPosition; // Get a start position.
201  float lerpTimer = 0.0f; // Make a timer.
202 
203  // While moving, update the position with a lerp.
204  while (lerpTimer < time)
205  {
206  lerpTimer += Time.deltaTime;
207  moveRTR.anchoredPosition = Vector2.Lerp(start, position, lerpTimer / time);
208  yield return null;
209  }
210  }
