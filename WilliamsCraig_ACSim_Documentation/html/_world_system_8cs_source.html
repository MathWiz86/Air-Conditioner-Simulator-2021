155  /// <summary>
156  /// The internal toggle function for if temperature fluctuation is active.
157  /// </summary>
158  /// <param name="isOn"></param>
159  [MethodImpl(MethodImplOptions.AggressiveInlining)]
160  private void ToggleActiveTemperatureFluctuationInternal(bool isOn)
161  {
162  isFluctuationActive = isOn;
163 
164  if (AllowTemperatureFluctuation)
165  {
166  // If fluctuation is active, and we're allowing fluctuation, turn on fluctuation.
167  if (isFluctuationActive && coFluctuation == null)
168  coFluctuation = StartCoroutine(FluctuateTemperature());
169  // Otherwise, if currently running, stop fluctuation.
170  else if (!isFluctuationActive && coFluctuation != null)
171  {
172  StopCoroutine(coFluctuation);
173  coFluctuation = null;
174  }
175  }
176  else if (coFluctuation != null)
177  {
178  StopCoroutine(coFluctuation);
179  coFluctuation = null;
180  }
181  }

204  /// <summary>
205  /// An IEnumerator for flucutating the world temperature at random.
206  /// </summary>
207  /// <returns>Returns an <see cref="IEnumerator"/> to use in <see cref="Coroutine"/>s.</returns>
208  private IEnumerator FluctuateTemperature()
209  {
210  while (true)
211  {
212 
213  // Get a wait timer based on the random range, and wait that many seconds.
214  float waitTimer = Random.Range(_FluctuationWaitRange.x, _FluctuationWaitRange.y);
215  yield return new WaitForSeconds(waitTimer);
216 
217  float direction = Random.Range(0, 2) == 1 ? 1 : -1; // Get if we are increasing or decreasing temperature.
218 
219  // Get the fluctuation per second, based on the random range.
220  float changePerSecond = direction * Random.Range(_FluctuationTemperatureRange.x, _FluctuationTemperatureRange.y);
221  // Get the amount of time to fluctuate, based on the random range.
222  float changeTimer = Random.Range(_FluctuationLengthRange.x, _FluctuationLengthRange.y);
223 
224  // While still changing, update the world temperature.
225  while (changeTimer > 0.0f)
226  {
227  changeTimer -= Time.deltaTime;
228  ChangeWorldTemperature(Time.deltaTime * changePerSecond);
229  yield return null;
230  }
231  }
232  }
