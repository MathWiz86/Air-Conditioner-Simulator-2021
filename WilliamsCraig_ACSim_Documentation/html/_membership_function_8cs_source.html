35  /// <summary>

51  /// If the function is not valid, returns <see cref="InvalidFunctionValue"/>.</param>
52  /// <returns>Returns if the function is valid, and thus returned a proper output.</returns>
53  public bool GetValueFromFunction(float crispInput, out float foundValue)
54  {

56  return isValidFunction;

65
