54  protected override float GetValueFromFunctionInternal(float crispInput)
55  {
56  switch (crispInput)
57  {
58  case float x when a == b && a == x: // Special case. If all values equal each other, return the max value.
59  return MaxValue;
60  case float x when b == c && b == x: // Special case. If all values equal each other, return the max value.
61  return MaxValue;
62  case float x when a <= x && x < b: // First branch
63  return (x - a) / (b - a);
64  case float x when b <= x && x < c: // Second branch
65  return (c - x) / (c - b);
66  default: // Default return
67  return MinValue;
68  }
69  }
