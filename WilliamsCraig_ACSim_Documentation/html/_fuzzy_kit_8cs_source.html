26  /// <summary>
27  /// A function which calculates the output of a Takagi-Sugeno fuzzy system, given that there is only one input.
28  /// </summary>
29  /// <param name="crispInput">The crisp input for the membership functions.</param>
30  /// <param name="inputFunctions">All membership functions that make up this system.</param>
31  /// <returns>Returns the final output value of the system.</returns>
32  public static float CalculateTakagiSugenoSingle(float crispInput, params TSMembershipFunction[] inputFunctions)
33  {
34  // Return an invalid value if there are no input functions.
35  if (inputFunctions.IsEmptyOrNull())
36  return MembershipFunction.InvalidFunctionValue;
37 
38  float topSum = 0.0f; // The top of the final equation.
39  float bottomSum = 0.0f; // The bottom of the final equation.
40 
41  int count = inputFunctions.Length;
42 
43  // Iterate through all input functions.
44  for (int i = 0; i < count; i++)
45  {
46  float fireStrength = inputFunctions[i].GetValueFromFunction(crispInput); // Get the firing strength.
47  float ruleOutput = inputFunctions[i].GetOutputFunctionValue(fireStrength); // Get the output of the rule.
48 
49  topSum += (fireStrength * ruleOutput); // The top is the sum of all firing strengths multiplied by the rule output.
50  bottomSum += fireStrength; // The bottom is the sum of all firing strengths.
51  }
52 
53  return topSum / bottomSum; // Return the final value.
54  }
55  }
56 }
