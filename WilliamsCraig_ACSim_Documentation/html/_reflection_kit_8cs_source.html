44  /// <summary>
45  /// A function for getting the value of a dynamically accessed field. This is useful for PropertyDrawers.
46  /// </summary>
47  /// <typeparam name="T">The type stored in the wanted value.</typeparam>
48  /// <param name="obj">The starting object to search from.</param>
49  /// <param name="path">The path to the value.</param>
50  /// <param name="default_return">The default value to return should the search fail.</param>
51  /// <param name="found_value">The found value at the path.</param>
52  /// <returns>Returns if the wanted value was successfully found.</returns>
53  public static bool CheckDynamicField<T>(object obj, string path, T default_return, out T found_value)
54  {
55  found_value = default_return; // Initialize the found value.
56 
57  // Check that nothing is empty, and that we are searching for a variable call.
58  if (!string.IsNullOrWhiteSpace(path) && path.First().ToString() == VariableCall)
59  {
60  string sub_path = path.Substring(1); // Get the substring without the variable call character.
61 
62  // Get the wanted field info.
63  object current = obj;
64  FieldInfo info = GetFieldInfo(ref current, out object previous, DefaultFlags, sub_path.Split('.'));
65 
66  // If the field info is found, check if it is of the wanted type. If so, return it.
67  if (info != null)
68  {
69  object value = info.GetValue(previous);
70  if (value is T)
71  {
72  found_value = (T)value;
73  return true;
74  }
75  }
76 
77  current = obj;
78  PropertyInfo pinfo = GetPropertyInfo(ref current, out previous, DefaultFlags, sub_path.Split('.'));
79 
80  if (pinfo != null)
81  {
82  object value = pinfo.GetValue(previous);
83  if (value is T)
84  {
85  found_value = (T)value;
86  return true;
87  }
88  }
89  }
90 
91  return false; // No value was found.
92  }

94  /// <summary>
95  /// A helper function to turn a group of individual paths into one formatted like a Unity Property Path.
96  /// </summary>
97  /// <param name="paths">The paths to concatenate. These are put together using the '.' character.</param>
98  /// <returns>Returns the full string of the paths concatenated.</returns>
99  public static string ConcatenatePaths(params string[] paths)
100  {
101  // Make sure the paths exist.
102  if (!paths.IsEmptyOrNullGeneric())
103  {
104  StringBuilder fullPath = new StringBuilder(); // Make a new StringBuilder.
105 
106  // Append each path, along with the path separator.
107  foreach (string p in paths)
108  fullPath.Append(p).Append(PathSeparator);
109 
110  fullPath.Remove(fullPath.Length - 1, 1); // Remove the final separator.
111  return fullPath.ToString(); // Return the full path.
112  }
113 
114  return string.Empty; // Return an empty string.
115  }

230  /// Make sure the the Field is serialized for custom Editors.
231  /// </summary>
232  /// <typeparam name="T">The type of the variable being looked for. If you don't know, use 'object'.</typeparam>
233  /// <param name="obj">The starting object to get the value off from.</param>
234  /// <param name="value">The value to set into the object.</param>
235  /// <param name="flags">The flags used to get the value. Use this to limit or expand what is available to access.</param>
236  /// <param name="path">The series of paths to get to the value. Unity concatonates all of these with a '.'.</param>
237  /// <returns>Returns if the value was properly set or not.</returns>
238  public static bool SetFieldValue<T>(object obj, T value, BindingFlags flags, params string[] path)
239  {
240  object current = obj; // The current object. This is a constantly changing reference.
241 
242  // Get the FieldInfo at the end of the series of paths. We will need the previous object.
243  FieldInfo field = GetFieldInfo(ref current, out object previous, flags, path);
244 
245  // Make sure the field is not null.
246  if (field != null)
247  {
248  // Switch on previous' type. This also allows for extra types later if required.
249  switch (previous)
250  {
251  // If the prevous object is an IList (IList<T> is an IList), and the last path points to an array element.
252  case IList ilist when path.LastElement().Contains(DataKeyword):
253  int index = ParseArrayPathIndex(path.LastElement()); // Parse the index out from the last path.
254 
255  // Check that the collection is not null, and the requested index is valid.
256  if (ilist != null && ilist.IsValidIndexGeneric(index))
257  {
258  ilist[index] = value; // Set the value at the requested index.
259  return true; // The value was successfully set.
260  }
261 
262  return false; // The value was not set.
263  default:
264  field.SetValue(previous, value);
265  return field.GetValue(previous) == (object)value; // Return if the value was correctly set.
266  }
267  }
268 
269  return false; // The value was not set.
270  }

285  /// <summary>
286  /// An internal function to acquire an object's FieldInfo, given a series of paths.
287  /// </summary>
288  /// <param name="current">The current object being looked into. This will end as the final value in the FieldInfo.</param>
289  /// <param name="previous">The previous object looked into.</param>
290  /// <param name="flags">The BindingFlags used to determine what values are allowed to be accessed.</param>
291  /// <param name="path">The array of path pieces to the wanted value. In Unity, this would be the property path, split by the '.' character.</param>
292  /// <returns>Returns the FieldInfo at the end of the paths. Returns null in the event of an error.</returns>
293  public static FieldInfo GetFieldInfo(ref object current, out object previous, BindingFlags flags, params string[] path)
294  {
295  FieldInfo field = null; // The FieldInfo of the field.
296  previous = current; // Set the previous object to null to start.
297  try
298  {
299  // Go through all the individual paths.
300  for (int i = 0; i < path.Length; i++)
301  {
302  // Switch on current's type. This also allows for extra types later if required.
303  switch (current)
304  {
305  // If the current object is an IList (IList<T> is an IList)
306  case IList ilist:
307  // Check if the current path matches the ArrayKeyword.
308  if (path[i] == ArrayKeyword)
309  {
310  i++; // The next path is guaranteed to be the element path.
311  int index = ParseArrayPathIndex(path[i]); // Parse out the index.
312  previous = current; // Set the previous object.
313  current = ilist.IsValidIndexGeneric(index) ? ilist[index] : null; // If the index is valid, return that object. Else, return null;
314  }
315  break;
316  default:
317  field = current.GetType().GetField(path[i], flags); // Get the field of the current object's type, using the given flags.
318  previous = current; // Update the previous object reference.
319  current = field.GetValue(current); // Update the current object reference with the current value.
320  break;
321  }
322  }
323  }
324  catch
325  {
326  // In the event of an error, return null for everything.
327  current = null;
328  previous = null;
329  return null;
330  }
331 
332  return field; // Return the found FieldInfo.
333  }
